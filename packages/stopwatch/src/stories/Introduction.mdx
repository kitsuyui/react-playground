## Role

This package provides stopwatch state/control primitives plus a replaceable default UI.

## Layers

- `StopwatchContextProvider` owns elapsed time state and control actions.
- `DefaultStopwatch` is the package-provided default UI on top of that primitive.

## Example

```tsx
import {
  DefaultStopwatch,
  StopwatchContext,
  StopwatchContextProvider,
  type StopwatchContextValue,
} from '@kitsuyui/react-stopwatch'

function StopwatchExample() {
  return (
    <StopwatchContextProvider>
      <StopwatchContext.Consumer>
        {(context: StopwatchContextValue) => <DefaultStopwatch {...context} />}
      </StopwatchContext.Consumer>
    </StopwatchContextProvider>
  )
}
```
